apt install docker.io -y
sudo apt install openjdk-11-jre -y
sudo apt install openjdk-11-jdk-headless -y
sudo apt install openjdk-11-jresudo -y
apt install gradle -y 
echo "Start."
cd /home/nikita/script
git clone https://gitlab.com/romanov73/example-jpa.git
cd ./example-jpa/
gradle build
docker build -t docker1 -f DockerBuild .
docker network create --driver bridge postgres-network
sudo docker cp /home/nikita/script/example-jpa/build/libs/example-jpa-0.1.0-SNAPSHOT.jar docker1:/var/
docker run --name docker1 --network postgres-network -v pgdata:/var/lib/postgresql/9.3/main -p 5432:5432 -d docker1
sudo docker exec -it docker1 java /var/example-jpa-0.1.0-SNAPSHOT.jar
echo "Done."
